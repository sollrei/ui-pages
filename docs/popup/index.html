Popup

Popper

<button class="ui-button js-popper">Click Me</button>

JavaScript

var btn = document.querySelector('.js-popper');
btn.addEventListener('click', function (e) {
  e.stopPropagation();
  
  new Popper(btn, {
    position: 'center top',
    content: '内容部分'
  });
}, false);
